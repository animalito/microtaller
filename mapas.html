Aadir variables al mapa

Nuestro objetivo final es hacer mapas para representar una variable, veamos como haramos para representar el ndice de carencias que creamos en la tarea.

library(Hmisc)

conapo <- read.csv("data/indice_carencias.csv", stringsAsFactors=FALSE)

conapo$CVE <- read.table("data/conapo_2010.csv", header=TRUE, quote="\"", 
  stringsAsFactors=FALSE, colClasses = "character")$CVE

# Creamos una variable categrica
conapo <- mutate(conapo, indice_cat = cut2(PC1, 5))
indice <- select(conapo, indice = PC1, indice_cat, CVE)

centro <- indice$CVE[conapo$NOM_ABR %in% c("DF", "Mex.", "Mor.")]

mun_shp <- readOGR("shapes" , "municipios_ligero")
## OGR data source with driver: ESRI Shapefile 
## Source: "C:\Users\animalito\Documents\learn\microtaller\shapes", layer: "municipios_ligero"
## with 2456 features
## It has 5 fields
## Integer64 fields read as strings:  OBJECTID
names(mun_shp)
## [1] "OBJECTID" "CVE_ENT"  "CVE_MUN"  "NOM_MUN"  "id"
mun_shp@data$CVE <- paste(as.character(mun_shp@data$CVE_ENT), 
  as.character(mun_shp@data$CVE_MUN), sep = "")

mun_df <- fortify(mun_shp, region = "CVE")

Para incluirlas en el mapa aadimos las variables de inters a la base de datos mun_df. Para este ejemplo graficaremos nicamente una regin. Podemos crear el subconjunto directamente del objeto SpatialPolygonsDataFrame pero en este ejemplo filtraremos la base de datos data.frame.

indice <- indice %>% mutate(CVE = stringr::str_pad(CVE, 5, side = "left", pad = "0"))
mun_ind <- mun_df %>%
  mutate(CVE = id) %>%
  left_join(indice) %>% 
  filter(CVE %in% centro)

ggplot() + 
  geom_polygon(data = mun_ind, aes(long, lat, group = group, fill = indice)) +
  labs(title = "ndice de carencias", fill = "ndice") +
  coord_fixed()

En el siguiente mapa los colores son a nivel municipio pero dibujamos las fronteras a nivel estado para evitar que los bordes opaquen los datos.

mun_shp_2 <- spTransform(mun_shp, CRS("+proj=longlat +datum=WGS84"))

mun_df <- fortify(mun_shp_2, region = "CVE")
mun_ind <- mun_df %>%
  mutate(CVE = id) %>%
  left_join(indice)

ggplot() + 
  geom_polygon(data = mun_ind, aes(long, lat, group = group, fill = indice)) +
